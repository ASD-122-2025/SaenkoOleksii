def bin_search(arr, target):
    left, right = 0, len(arr) - 1
    while left <= right:
        mid = (left + right) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            left = mid + 1
        else:
            right = mid - 1
    return -1
def create_dataframe(max_n, step):
    data = []

        arr = sorted(random.sample(range(1, 10 * n), n))
        target = random.choice(arr)
        start_time = time.time()
        bin_search(arr, target)
        end_time = time.time()
        data.append({"n": n, "time": end_time - start_time})
    df = pd.DataFrame(data)
    return df
